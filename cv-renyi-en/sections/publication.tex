        \begin{onecolentry}
            \textbf{Yi Ren} (arXiv preprint \href{https://arxiv.org/abs/2605.25781}{arXiv:2605.25781}, 2026)
        \end{onecolentry}
        \vspace{0.10 cm}

        \begin{onecolentry}
        \begin{highlights}
            \item Designed an annotation framework around a consensus principle, applied recursively at two layers---between two independent MLLMs (Claude, Qwen, Llama, Grok), then between two human-in-the-loop systems built from them---escalating only disagreements to a human jury
            \item Achieved a Word Error Rate of 0.003 on the Rosenwald Guides, with >85\% of fields auto-accepted by model consensus---cutting total human review workload by more than 50\% compared to a full manual annotation by one annotator
            \item Released the first structured-extraction benchmark for the Rosenwald Guides (13,595 annotated fields across 60 columns) to support future work on historical document processing
        \end{highlights}
        \end{onecolentry}

\textbf{MAM-AI: An On-Device Medical Retrieval-Augmented Generation System for Nurses and Midwives in Zanzibar} [\href{https://arxiv.org/abs/2606.29580}{Preprint}] [\href{https://github.com/nmrenyi/mamai}{Code}] [\href{https://github.com/nmrenyi/mamai-eval}{Eval}] [\href{https://nmrenyi-mamai-demo.hf.space}{Live Demo}]

\end{twocolentry}

        \vspace{0.10 cm}
        \begin{onecolentry}
            \textbf{Yi Ren} (arXiv preprint \href{https://arxiv.org/abs/2606.29580}{arXiv:2606.29580}, 2026)
        \end{onecolentry}
        \vspace{0.10 cm}

        \begin{onecolentry}
        \begin{highlights}
            \item A redesigned system prompt is a major lever on answer quality---cutting deflective answers that only said ``refer to a senior doctor'' from 32.7\% of questions to 3.2\%, while roughly doubling key-fact recall from 0.139 to 0.279 and raising potentially-harmful answers only from 12.8\% to 15.7\%
            \item On-device retrieval is strong, but ranking is not the lever: on our retrieval benchmark the 300M on-device embedder ranks among the cloud retrievers, yet a better embedder or a fine-tuned reranker leaves end-to-end answers unchanged---the larger levers on answer quality are the generator and corpus coverage and quality
            \item The generator matters most, and faithfulness decides deployment: answer quality rises with model size before plateauing, while quantization and retrieval barely move it; among two same-size on-device candidates, the more faithful but less helpful Gemma 4 E4B is deployed over Gemma 3n E4B---a safety-versus-usefulness choice
        \end{highlights}
        \end{onecolentry}

\textbf{mamabench and mamaretrieval: Benchmarks for Evaluating Medical Retrieval-Augmented Generation in Maternal, Neonatal, and Reproductive Health} [\href{https://arxiv.org/abs/2606.29467}{Preprint}] [\href{https://huggingface.co/datasets/nmrenyi/mamabench}{mamabench}] [\href{https://huggingface.co/datasets/nmrenyi/mamaretrieval}{mamaretrieval}]

\end{twocolentry}

        \vspace{0.10 cm}
        \begin{onecolentry}
            \textbf{Yi Ren} (arXiv preprint \href{https://arxiv.org/abs/2606.29467}{arXiv:2606.29467}, 2026)
        \end{onecolentry}
        \vspace{0.10 cm}

        \begin{onecolentry}
        \begin{highlights}
            \item Assembled both benchmarks---mamabench (25,949 QA items) and mamaretrieval (3,185 retrieval queries)---by scoping and filtering expert-authored sources (licensing exams, physician panels, guidelines) rather than authoring new clinical questions
            \item Decomposed clinical relevance into a graded, multi-dimensional rubric (0--6)---a topical gate over three graded dimensions: meaningful content, actionable guidance, and density---rather than a binary relevant/not-relevant label, separating strong retrieval from merely adequate retrieval
            \item Measured and disclosed the limits of the LLM-assigned labels---via scope-classifier agreement, a frontier-judge check, and a pooling-completeness audit---rather than presenting the benchmarks as an oracle
        \end{highlights}
        \end{onecolentry}
